## 📊 СТРУКТУРА ДАННЫХ

### База данных схема

```sql
-- Пользователи
CREATE TABLE users (
    id UUID PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(255),
    subscription_tier VARCHAR(50), -- free, basic, premium, super
    subscription_expires_at TIMESTAMP,
    created_at TIMESTAMP,
    last_login TIMESTAMP,
    settings JSONB -- темная тема, язык, etc
);

-- Уроки
CREATE TABLE lessons (
    id SERIAL PRIMARY KEY,
    title VARCHAR(500),
    slug VARCHAR(500),
    content TEXT,
    category VARCHAR(100),
    order_index INT,
    is_free BOOLEAN DEFAULT false,
    images JSONB,
    created_at TIMESTAMP
);

-- Вопросы
CREATE TABLE questions (
    id SERIAL PRIMARY KEY,
    question_text TEXT NOT NULL,
    question_image_url VARCHAR(500),
    options JSONB, -- [{text: "A", is_correct: true}, ...]
    explanation TEXT,
    article_reference VARCHAR(100), -- ссылка на статью кодекса
    category VARCHAR(100),
    difficulty VARCHAR(50), -- easy, medium, hard
    created_at TIMESTAMP
);

-- Прогресс пользователя
CREATE TABLE user_progress (
    id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    lesson_id INT REFERENCES lessons(id),
    completed BOOLEAN DEFAULT false,
    completed_at TIMESTAMP,
    time_spent INT -- в секундах
);

-- Результаты тестов
CREATE TABLE test_attempts (
    id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    test_type VARCHAR(50), -- practice, exam, marathon
    questions JSONB, -- [{question_id, user_answer, is_correct}]
    score INT, -- правильных ответов
    total_questions INT,
    started_at TIMESTAMP,
    finished_at TIMESTAMP,
    time_taken INT -- в секундах
);

-- Оплаты
CREATE TABLE payments (
    id UUID PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    amount DECIMAL(10,2),
    currency VARCHAR(3),
    status VARCHAR(50), -- pending, completed, failed, refunded
    payment_provider VARCHAR(50), -- stripe, mollie
    subscription_tier VARCHAR(50),
    created_at TIMESTAMP
);

-- Дорожные знаки
CREATE TABLE road_signs (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255),
    category VARCHAR(100), -- warning, prohibition, mandatory, etc
    image_url VARCHAR(500),
    description TEXT,
    official_code VARCHAR(50)
);

-- Статьи кодекса
CREATE TABLE code_articles (
    id SERIAL PRIMARY KEY,
    article_number VARCHAR(50),
    title VARCHAR(500),
    content TEXT,
    category VARCHAR(100),
    is_free BOOLEAN DEFAULT false
);
```
